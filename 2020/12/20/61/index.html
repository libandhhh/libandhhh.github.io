    static BufferedWriter out;

    public static void main(String[] args) throws IOException {
        //StreamTokenizer in = new StreamTokenizer(new BufferedReader(new InputStreamReader(new FileInputStream("2020.txt"))));
        out = new BufferedWriter(new OutputStreamWriter(System.out));
        Scanner in = new Scanner(System.in);
        int t = in.nextInt();
        int m = in.nextInt();
        int weight[] = new int[m], value[] = new int[m];
        int dp[] = new int[t + 1];
        for (int i = 0; i < m; i++) {
            weight[i] = in.nextInt();
            value[i] = in.nextInt();
        }
        for (int i = 0; i < m; i++) {
            for (int j = t; j >= 0; j--) {
                if (j >= weight[i]) {
                    dp[j] = Math.max(dp[j], dp[j - weight[i]] + value[i]);
                }
            }
        }
        out.write(String.valueOf(dp[t]));
        out.flush();
    }
